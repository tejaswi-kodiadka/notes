\chapter{Matrix Groups}

\section{Groups of Matrices}

Let \(M_{m,n}(\K)\) be the set of \(m\times n\) matrices whose entries are in \(\K\), a commutative field.
We will also use the special notations \(M_n(\K)=M_{n.n}(\K)\), and \(\K^n=M_{n,1}(\K)\).

\(M_{m,n}(\K)\) is a vector space over \(\K\) with dimension \(mn\). As well as being a vector space over
\(\K\) with dimension \(n^2\), \(M_n(\K)\) is also a ring with the usual addition and multiplication of
matrices. It is also an important example of a finite dimensinoal algebra over \(\K\).

We will use the notation
\[ GL_n(\K)=\{A\in M_n(\K):\text{det }A\ne0\} \]
for the set of invertible \(n\times n\) matrices, known as the \(n\times n\) {\it general linear group}, and 
\[ SL_n(\K)=\{A\in M_n(\K):\text{det }A=1\}\subseteq GL_n(\K) \]
for the set of \(n\times n\) unimodular matrices, known as the \(n\times n\) {\it special linear group}.

\begin{theorem}
  The sets \(GL_n(\K)\) and \(SL_n(\K)\) are groups under matrix multiplication. Furthermore,
  \(SL_n(\K)\) is a subgroup of \(GL_n(\K)\).
\end{theorem}

Since determinant of a matrix is a polynomial in the entries of the matrix, it can
be proved that the determinant function
\[\text{det}:M_n(\K)\longrightarrow\K\]
is continous.

\begin{proposition}
  Let \(\K=\R\) or \(\C\). Then
  \begin{itemize}
    \item[\((i)\)] \(GL_n(\K)\subseteq M_n(\K)\) is an open subset;
    \item[\((ii)\)] \(SL_n(\K)\subseteq M_n(\K)\) is a closed subset.
  \end{itemize}
\end{proposition}

\begin{proof}
  \(GL_n(\K)=M_n(\K)-\text{det}^{-1}\{0\}\), which is open since \{0\} is closed. Similarly,
  \(SL_n(\K)=\text{det}^{-1}\{1\}\), which is closed since the singleton \{1\} is closed in \(\K\).
\end{proof}

\begin{definition}
  Let \(G\) be a topological space, and view \(G\times G\) as the product space. Suppose that
  \(G\) is also a group with multiplication map \(\text{mult }:G\times G\longrightarrow G\),
  and inverse map \(\text{inv }:G\longrightarrow G\). \(G\) is a {\it topological group} if
  mult and inv are continous.
\end{definition}

\begin{theorem}
  For \(\K=\R\) or \(\C\), \(GL_n(\K)\) and \(SL_n(\K)\) are topological groups.
\end{theorem}

\section{Matrix Groups}

\begin{definition}
  A subgroup \(G\) of \(GL_n(\K)\) is called a {\it matrix group} if it is a closed subset of \(GL_n(\K)\).
\end{definition}

\begin{remark}
  We may consider \(GL_n(\K)\) as a subgroup of \(GL_{n+1}(\K)\) by identifying the \(n\times n\)
  matrix \(A=[a_{ij}]\) with
  \[
    \left[\begin{array}{cc}
      A & 0 \\
      0 & 1
    \end{array}\right]
    =
    \left[\begin{array}{cccc}
      a_{11} & \cdots & a_{1n} & 0 \\
      \vdots & \ddots & \vdots & \vdots \\
      a_{n1} & \cdots & a_{nn} & 0 \\
      0 & \cdots & 0 & 1
    \end{array}\right]
  \]
  And it is easily verified that \(GL_n(\K)\) is closed in \(GL_{n+1}(\K)\). With the aid of this embedding, any matrix subgroup of \(GL_n(\K)\) can also be viewed as
  a matrix subgroup of \(GL_{n+1}(\K)\).
\end{remark}

\begin{definition}
  Let \(G\), \(H\) be two matrix groups. A groups homomorphism \(\varphi:G\longrightarrow H\) is a
  {\it continuous homomorphism of matrix groups} if it is continuous and its image
  \(\varphi(G)\) is a closed subgroup of \(H\).
\end{definition}

\begin{examples}
  Some important examples of real and complex matrix groups:
  \begin{enumerate}
    \item {\bf Groups of Upper Triangular Matrices}

      For \(n\ge1\), an \(n\times n\) matrix \(A=[a_{ij}]\) is said to be {\it upper triangular}
      if \(a_{ij}=0\) if \(i<j\). An upper traingular matrix is said to be {\it unipotent} of all
      of it's diagonal entries are 1.

      The {\it upper triangular subgroup} or {\it Borel subgroup} of \(GL_n(\K)\) is
      \[ UT_n(\K)=\{A\in GL_n(\K):\text{A is upper triangular}\}, \]
      while the {\it unipotent} subgroup of \(GL_n(\K)\) is
      \[ SUT_n(\K)=\{A\in GL_n(\K):\text{A is unipotent}\}. \]

      It is easy to verify that \(UT_n(\K)\) and \(SUT_n(\K)\) are closed subgroups of \(GL_n(\K)\),
      and that \(SUT_n(\K)\) is a subgroup of \(UT_n(\K)\).
    \item {\bf Affine Groups}

      The {\it n-dimensional affine group} over \(\K\) is
      \[ Aff_n(\K) = \left\{ \left[\begin{array}{cc} A & t \\ 0 & 1 \end{array}\right]:A\in GL_n(\K), t\in \K^n\right\} \]
      which is clearly a closed subgroup of \(GL_{n+1}(\K)\). If we identify \(x\in\K^n\) with
      \(\left[\begin{array}{c} x \\ 1 \end{array}\right]\in\K^{n+1}\), then as a consequence of
      \[ 
        \left[\begin{array}{cc}
          A & t \\
          0 & 1 
        \end{array}\right]
        \left[\begin{array}{c} x \\ 1 \end{array}\right]
        =
        \left[\begin{array}{c} Ax+t \\ 1 \end{array}\right]
      \]
      we obtain an action of \(Aff_n(\K)\) on \(\K^n\).

      The group \(Aff_n(\K)\) can be expressed as the semi-direct product of \(Trans_n(\K)\)
      and \(GL_n(\K)\), where
      \[ Trans_n(\K) = \left\{ \left[\begin{array}{cc} I_n & t \\ 0 & 1 \end{array}\right]:t\in \K^n\right\} \]
      is a closed normal subgroup of \(Aff_n(\K)\). That is,
      \[ Aff_n(\K)=GL_n(\K)\ltimes Trans_n(\K). \]

    \item {\bf Orthagonal Groups}

      The {\it \(n\times n\) real orthagonal group} is the subset
      \[ O(n)=\{A\in GL_n(\R):A^TA=I_n \}, \]
      where \(A^T\) is the transpose of \(A=[a_{ij}]\) whose entris are given by
      \((A^T)_{ij}=a_{ji}\). \(O(n)\) is a closed subgroup of \(GL_n(\R)\). For all \(A\in O(n)\),
      since \(\text{det } A= \text{det } A^T\), \(\text{det } A= \pm 1\). The subgroup
      \[ SO(n)=\{A\in O(n):\text{det } A=1 \} \]
      is known as the {\it special orthagonal group}. One of the main reasons for
      the study of orthagonal groups is due to their relation with isometries. If \(A\in GL_n(\R)\),
      then it can be proved that \(A\) is an linear isometry {\it iff} \(A\) is orthagonal.
      The elements of \(SO(n)\) are called {\it direct isometries} or {\it rotations}.

    \item {\bf Unitary Groups}

      The {\it \(n\times n\) unitary group} is the subset
      \[ U(n)=\{A\in GL_n(\C):A^*A=I_n \}, \]
      where \(A^*\) is the {\it hermitian conjugate} of \(A\), that is \(A^*)_{ij}=a_{ji}\).
      \(U(n)\) is clearly a closed subgroup of \(GL_n(\C)\). The {\it special unitary group}
      is the subset
      \[ SU(n)=\{A\in U(n):\text{det } A=1 \}, \]
      which is also a normal subgroup of \(U(n)\).

    \item {\bf Symplectic Groups}

      A matrix \(S\) is called {\it skew symmetric} if \(S^T=-S\). The standard example of
      this is built up using the \(2\times 2\) block
      \[
        J=
        \left[\begin{array}{cc}
            0 & 1 \\
            -1 & 0
        \end{array}\right].
      \]
      When \(m\ge1\) we have the non degenrative skew symmetric matrix
      \[
        J_{2m}=
        \left[\begin{array}{cccc}
            J & O_2 & \cdots & O_2 \\
            O_2 & J & \cdots & O_2 \\
            \vdots & \vdots & \ddots & \vdots \\
            O_2 & O_2 & \cdots & J
        \end{array}\right].
      \]

      The matrix group
      \[ Symp_{2m}(\R)=\{A\in GL_{2m}(\R): A^TJ_{2m}A=J_{2m} \} \]
      is called the {\it \(2m\times 2m\) real symplectic group}. It can be proved that if
      \(A\in Symp_{2m}(\R)\), then \(\text{det } A=1\).

  \end{enumerate}
\end{examples}

\section{Complex Matrices as Real Matrices}

Similar to how complex numbers can be viewed as a 2-dimensional real vector space with
basis \((1,i)\), every \(n\times n\) complex matrix can also be viewed as a \(2n\times 2n\)
real matrix using the following constructions or the ones similar to it.

We identify each complex number \(z=x+iy\) with a \(2\times 2\) matrix by defining a funciton
\[ \rho :\C\longrightarrow M_2(\R);\hspace{10px} \rho(x+iy) = \left[\begin{array}{cc} x & -y \\ y & x \end{array}\right]. \]
This can also be expressed as
\[ \rho(x+iy) = xI_2 - yJ. \]
As \(\rho\) is an injective ring homomorphism, we can view \(\C\) as a subring of \(M_2(\R)\).
Notice that \(\rho(\bar{z})=\rho(z)^T\).

For a \(n\times n\) complex matrix \(Z\), we define a function \(\rho_n: M_n(\C)\longrightarrow M_{2n}(\R)\);
\[
  \rho_n(Z)=
  \left[\begin{array}{cccc}
      \rho(z_{11}) & \rho(z_{12}) & \cdots & \rho(z_{1n}) \\
      \rho(z_{21}) & \ddots & \ddots & \vdots \\
      \vdots & \ddots & \ddots & \vdots \\
      \rho(z_{n1})& \cdots & \cdots & \rho(z_{nn})
  \end{array}\right]
\]
Then \(\rho_n\) is an continuous injective ring homomorphism. Generally for \(n\times n\)
real matrices \(X\) and \(Y\),
\[ \rho_n(X+iY)=\rho_n(X)-J_{2n}\rho_n(Y)J_{2n}. \]

\section{Matrix Groups of Normed Vector Spaces}

It is often desireable to generalize the notion of matrix spaces to srbitrary finite dimensional
real or complex vector spaces. Here we assume \(\K=\R\text{ or }\C\).

\begin{theorem}
  Let \((U,\mu)\) and \((V,\nu)\) be two finite dimensional normed vector spaces of the
  same dimension over \(\K\). Then any linear isomorphism \(\varphi:U\longrightarrow V\) is
  a homeomorphism.
\end{theorem}

\begin{corollary}
  If \(\mu\) and \(\nu\) are to norms on a finite dimensional vector space, then they give
  raise to the same topology.
\end{corollary}

\begin{proof}
  The identity function is a linear isomorphism, which must m=be continuous. This implies that
  every open set in the topology of \(\nu\) is open in the topology of \(\mu\) and vice-versa.
\end{proof}

Given a finite dimensional normed vector space \(V,\nu\) and a linear transformation
\(\alpha:V\longrightarrow V\), we define the {\it operator norm} of \(\alpha\) with respect to
\(\nu\) by
\[ ||\alpha||_{\nu}=sup\{\nu(\alpha(x)):x\in V,\nu(x)=1\} \]
This defines a norm on \(End_{\K}(V)\), the set of all linear transformations \(V\longrightarrow V\),
which makes it a finite dimensional normed vector space overe \(\K\).

\begin{proposition}
  If \(\mu\) and \(\nu\) are two norms on a finite dimensional vector space \(V\), then the
  topologies associated to \(End_{\K}(V)\) with respect to \(\mu\) and \(\nu\) are the same.
\end{proposition}

Inside \(End_{\K}(V)\) we have the group of linear isomorphisms, \(GL_{\K}(V)\subseteq End_{\K}(V)\),
the {\it general linear group} of the vector space \(V\).

\begin{proposition}
  Let \(U\) and \(V\) be two vector spaces of the same dimension. If \(\varphi:U\longrightarrow V\)
  is a linear isomorphism, then it induces a continuous group isomorphism with a continuous inverse,
  \[ \varphi_*:GL_{\K}(U)\longrightarrow GL_{\K}(V);\hspace{15px} \varphi_*(\alpha)=\varphi\circ\alpha\circ\varphi^{-1} \]
\end{proposition}

Generally, given a finite basis for \(V\), there is an associated linear isomorphism
\(V\cong\K^n\) for some \(n\), a ring isomorphism \(End_{\K}(V)\cong M_n(\K)\) and a group
isomorphism \(GL_{\K}(V)\cong GL_n(\K)\).

\begin{theorem}
  Let \(V\) be a finite dimensional vector space over \(\K\) with dimension \(n\) and let
  \(\varphi:V\longrightarrow\K^n\) be a linear isomorphism. Then \(\varphi_*:GL_{\K}(V)\longrightarrow GL_n(\K)\)
  is a continuous grouop isomorphism with a continuous inverse. Hence \(GL_{\K}(V)\) is a
  matrix group.
\end{theorem}

\section{Matrix Exponential and Logarithm}

We define the exponential and logarithm functions for matrices using theor power series.
Throught this section, we assume \(\K=\R\text{ or }\C\).
\\

The {\it exponential function} is, \(\exp:M_n(\K)\longrightarrow GL_n(\K)\);
\[ \exp(A)=\sum_{n\ge0}\frac{1}{n!}A^n=I+A+\frac{1}{2!}A^2+\frac{1}{3!}A^3+\cdots \]

\begin{proposition}
  Let \(A\in M_n(\K)\).
  \begin{itemize}
    \item[\((i)\)] For \(u,v\in\C, \exp((u+v)A)=\exp(uA)\exp(vA)\).
    \item[\((ii)\)] \(\exp(A)^{-1}=\exp(-A).\)
    \item[\((iii)\)] If \(B\in M_n(\K)\) commutes with \(A\), then \(\exp(A+B)=\exp(A)\exp(B)\).
  \end{itemize}
\end{proposition}

We define the {it logarithm function} by
\[ \log:N_{M_n(\K)}(I;1)\longrightarrow M_n(\K); \]
\[ \log(A)=\sum_{n\ge1}\frac{(-1)^{n-1}}{n}(A-I)^n \]

\begin{proposition}
  The \(\exp\) and \(\log\) functions satisfy
  \begin{itemize}
    \item[\((i)\)] if \(||A-I||<1\), then \(\exp(\log(A))=A\);
    \item[\((ii)\)] if \(||exp(B)-I||<1\), then \(\log(\exp(B))=B\).
  \end{itemize}
\end{proposition}

The functions \(\exp\) and \(\log\) are continuous and in fact infinitely differentiable
on their domains. By continuity of \(\exp\) at \(O\), there is a \(\delta_1>0\) such that
\[ N_{M_n(\K)}(O;r)\subseteq\exp^{-1}N_{GL_n(\K)}(I;1). \]
Since \(\exp N_{M_n(\K)}(O;r)\subseteq N_{M_n(\K)}(I;e^r-1)\), we can take \(\delta_1=\log 2\).
Hence we have the following result.

\begin{proposition}
  The exponential function \(\exp\) is injective when restricted to the open subset
  \(N_{M_n(\K)}(O;\log2)\subseteq M_n(\K)\). Hence it is locally a diffeomorphism at \(O\) with the
  local inverse \(\log\).
\end{proposition}

\section{Differential Equations in Matrices}

\begin{definition}
  A {\it differentiable curve} in \(M_n(\K)\) is a function \(\alpha:(a,b)\longrightarrow M_n(\K)\)
  for which the derivative \(\alpha'(t)\) exists for each \(t\in(a.b)\). Here \(\alpha'(t)\)
  is defined as an element of \(M_n(\K)\) by
  \[ \alpha'(t)=\lim_{s\rightarrow t}\frac{1}{(s-t)}(\alpha(s)-\alpha(t)), \]
  provided that the limit exists.
\end{definition}

Consider the first order differential equation \(\alpha'(t)=\alpha(t)A\), where \(\alpha\)
is assumed to be a differentiable curve.

\begin{theorem}
  For \(A,C\in M_n(\R)\) with \(A\) non-zero, and \(a<0<b\), the differential equation
  \(\alpha'(t)=\alpha(t)A\), has a unique solution \(\alpha:(a.b)\longrightarrow M_n(\R)\)
  for which \(\alpha(0)=C\). Furthermore, if \(C\) is invertible, then so is \(\alpha(t)\) for
  \(t\in(a,b)\).
\end{theorem}

\begin{proof}
Consider the function \(\alpha:(a,b)\longrightarrow M_n(\R)\); \(\alpha(t)=\exp(tA)\).
It is differentiable with
\[ \alpha'(t)=\sum{k\ge1}\frac{t^{k-1}}{(k-1)!}A^k=A\exp(tA) \]
Hence \(\alpha\) satisfies the differential equation with boundary condition \(\alpha(0)=I\).
Clearly, \(\alpha(t)=C\exp(tA)\) is one solution to \(\alpha(0)=0\). If \(\beta\) is another
such solution, then \(\gamma(t)=\beta(t)\exp(-tA)\) satisfies

\begin{eqnarray*}
  \gamma'(t) &=& \beta'(t)\exp(-tA)+\beta(t)\exp(-tA)(-A) \\
             &=& \beta(t)A\exp(-tA)-\beta(t)A\exp(-tA) \\
             &=& O.
\end{eqnarray*}

Hence \(\gamma\) is a constant function with \(\gamma(t)=\gamma(0)=C\). Thus
\(\beta(t)=C\exp(tA)=\alpha(t)\).
\end{proof}

\section{One Parameter Subgroups in Matrix Groups}
